//문자열을 정수형으로 변환하기(+charGPT: sizeof(char type array)는 array의 size가 아닌 pointer의 크기를 반환. 따라서 strlen로 사용) 

#include <iostream>
#include <string>
#include <cstring>
using namespace std;

int stringToInt(const char x[])
{
	for (int i = 0;i < strlen(x);i++)
	{
		if (x[i] >= '0' && x[i] <= '9')
			continue;
		else
			throw x;
	}
	int ret = stoi(x);
	return ret;
}

int main()
{
	char ch[5] = "Hi";
	char num[5] = "1234";

	try {
		cout << "숫자 문자열을 정수형으로 변경" << endl;
		int b = stringToInt(num);
		cout << "변경 결과: " << b << endl;

		cout << "문자 문자열을 정수형으로 변경" << endl;
		int a = stringToInt(ch);
		cout << "변경 결과: " << a << endl;
	}
	catch (const char* s)
	{
		cout << "숫자가 아닌 문자는 정수형으로 변경 불가" << endl;
	}
}
