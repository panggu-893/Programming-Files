		this->water = water; this->espresso = coffee;

		cout << "물 " << water << ", 커피 " << espresso;

{
	this->water += adder.water;
	this->espresso += adder.espresso;

	return *this;
}

{
	Coffee temp;
	temp = *this;
	temp.espresso += espresso;
	
	return temp;
}

#include <iostream>
#include <string>
using namespace std;

class Wallet
{
	string name;
	int balance;
public:
	Wallet(string name = "", int balance = 0) { this->name = name;this->balance = balance;}
	void show() { cout << name << " 지갑에 " << balance << "원 있어요." << endl; }
	Wallet operator += (int balance) { this->balance += balance; return *this; }
	Wallet operator -= (int balance) { this->balance -= balance; return *this; }
	Wallet& operator +(Wallet& w) { this->balance += w.balance; return *this; }
	Wallet& operator +(int n) { this->balance += n; return *this; }
	Wallet& operator = (Wallet& w) { this->balance = w.balance; return *this; }
	friend Wallet& operator + (int n, Wallet& w);
};

Wallet& operator + (int n, Wallet& w)
{
	w.balance += n;
	
	return w;
}


int main()
{
	Wallet a("옥지윤", 2000), b("성주연", 100), c("박채원");
	
	a += 1000;
	a.show();

	a -= 500;
	a.show();

	b = a + b + 100;
	b.show();

	c = 500 + b;
	c.show();
}

#include <iostream>
using namespace std;

class Light {
	int size;
public:
	Light(int size = 1) { this->size = size; }
	Light& operator += (Light& l) { this->size += l.size; l.size = 0; return *this; }
	Light& operator += (int n) { this->size += n; return *this; }
	void operator ~() { cout << "빛 크기 " << size << endl; }
	Light& operator = (const Light& l) { this->size = l.size; return *this; }
	Light operator --(int n) { Light temp; temp = *this; this->size--; return temp; }
	bool operator ==(int n) { if (this->size == n) return true; else return false; }
};



int main()
{
	Light a(10), b(20), c, d(0);
	(a += b) += 10;
	~a;~b;~c;
	d = c--;
	~d;
	if (c == 0) cout << "빛 c는 꺼져 있습니다." << endl;
	if (d == 0) cout << "빛 d는 꺼져 있습니다." << endl;
}

#include <iostream>
using namespace std;

class Matrix {
	int a1, a2, a3, a4;
public:
	Matrix(int a1 = 0, int a2 = 0, int a3 = 0, int a4 = 0) { this->a1 = a1;this->a2 = a2;this->a3 = a3;this->a4 = a4; }
	void show() { cout << "Matrix = { " << a1 << ' ' << a2 << ' ' << a3 << ' ' << a4 << " }" << endl; }
	Matrix operator + (Matrix& m) { Matrix temp; temp = *this; temp.a1 += m.a1; temp.a2 += m.a2; temp.a3 += m.a3; temp.a4 += m.a4; return temp; }
	Matrix& operator = (const Matrix& m) { this->a1 = m.a1;this->a2 = m.a2;this->a3 = m.a3;this->a4 = m.a4; return *this; }
	Matrix& operator += (Matrix& m) { this->a1 += m.a1;this->a2 += m.a2;this->a3 += m.a3;this->a4 += m.a4; return *this; }
	bool operator ==(Matrix& m) { if ((a1 == m.a1) && (a2 == m.a2) && (a3 == m.a3) && (a4 == m.a4)) return true; else return false; }
};

int main()
{
	Matrix a(1, 2, 3, 4), b(2, 3, 4, 5), c;
	c = a + b;
	a += b;
	a.show();b.show();c.show();
	if (a == c) cout << "a와 c는 같습니다." << endl;

	return 0;
}
